#include <stdio.h>
#include <stdlib.h>

int addition(int n1, int n2) {
    return n1 + n2;
}

int subtraction(int n1, int n2) {
    return n1 - n2;
}

int multiplication(int n1, int n2) {
    return n1 * n2;
}

int division(int n1, int n2) {
    if (n2 == 0) {
        printf("Cannot divide by zero!\n");
        return 0;
    }
    return n1 / n2;
}

int moduloDivision(int n1, int n2) {
    if (n2 == 0) {
        printf("Cannot perform modulo by zero!\n");
        return 0;
    }
    return n1 % n2;
}

int main() {
    int num1, num2, result, choice;

    while (1) {
        printf("Enter first number: ");
        scanf("%d", &num1);

        printf("Enter second number: ");
        scanf("%d", &num2);

        printf("\nMenu:\n");
        printf("1. Addition\n");
        printf("2. Subtraction\n");
        printf("3. Multiplication\n");
        printf("4. Division\n");
        printf("5. Modulo Division\n");
        printf("6. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                result = addition(num1, num2);
                printf("Sum is: %d\n", result);
                break;
            case 2:
                result = subtraction(num1, num2);
                printf("Difference is: %d\n", result);
                break;
            case 3:
                result = multiplication(num1, num2);
                printf("Product is: %d\n", result);
                break;
            case 4:
                result = division(num1, num2);
                printf("Division is: %d\n", result);
                break;
            case 5:
                result = moduloDivision(num1, num2);
                printf("Modulo is: %d\n", result);
                break;
            case 6:
                printf("Exiting...\n");
                exit(0);
            default:
                printf("Invalid choice. Try again.\n");
        }

        printf("\n--------------------------\n");
    }

    return 0;
}
